## Usage
<ExampleContainer>
    <Tabs filter={categoryFilter()} tabs={[{name: "All", display: "All"}, {name: "Blog", display: "Blog"}]} />
</ExampleContainer>

```javascript
import { Filter, Tabs } from "@sajari/sdk-react"

const filter = new Filter({
        "All": "",
        "Blog": "dir1='blog'",
    },
    ["All"]
);

<Tabs
    filter={filter}
    tabs={[
        {name: "All", display: "All"},
        {name: "Blog", display: "Blog"}
    ]}
/>
```
